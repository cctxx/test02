//Navigation mesh builder interface.
CLASS NavMeshBuilder

	// Object containing settings for the Navmesh
	//*undocumented*
	CUSTOM_PROP static UnityEngine.Object navMeshSettingsObject
	{
		return Scripting::ScriptingWrapperFor(&GetNavMeshSettings());
	}
	
	// Build the Navmesh.
	CUSTOM static void BuildNavMesh ()
	{
		NavMeshBuilder::BuildNavMesh(GetNavMeshSettings().GetNavMeshBuildSettings());
	}

	// Build the Navmesh Asyncronously.
	CUSTOM static void BuildNavMeshAsync ()
	{
		NavMeshBuilder::BuildNavMeshAsync(GetNavMeshSettings().GetNavMeshBuildSettings());
	}

	// Clear all Navmeshes.
	CUSTOM static void ClearAllNavMeshes ()
	{
		NavMeshBuilder::ClearAllNavMeshes();
	}

	// Returns true if an asynchronous build is still running.
	CUSTOM_PROP static bool isRunning
	{
		return NavMeshBuilder::IsBuildingNavMeshAsync();
	}

	// Cancel Navmesh construction.
	CUSTOM static void Cancel ()
	{
		return NavMeshBuilder::Cancel();
	}

END

CLASS internal NavMeshVisualizationSettings
	
	CUSTOM_PROP static bool showNavigation
	{ return GetNavMeshVisualizationSettings().GetShowNavigation(); }
	{ GetNavMeshVisualizationSettings().SetShowNavigation(value); }
	
	CUSTOM_PROP static bool showNavMesh

	{ GetNavMeshVisualizationSettings().SetShowNavMesh(value); }
	
	CUSTOM_PROP static bool showHeightMesh

	{ GetNavMeshVisualizationSettings().SetShowHeightMesh(value); }
	
	CUSTOM_PROP static bool hasHeightMesh
	{ return GetNavMeshVisualizationSettings().HasHeightMesh(); }

END
