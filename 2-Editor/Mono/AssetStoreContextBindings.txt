C++RAW


typedef std::map<string,string> SessionStrings;
static SessionStrings s_SessionStrings;


CLASS internal AssetStoreContext

CUSTOM public static void SessionSetString(string key, string value)
{
	s_SessionStrings[key] = value;
}

CUSTOM public static string SessionGetString(string key)
{
	SessionStrings::iterator i = s_SessionStrings.find(key);
	if (i == s_SessionStrings.end())
		return scripting_string_new("");
	return scripting_string_new(i->second);
}

CUSTOM public static void SessionRemoveString(string key)
{
	s_SessionStrings.erase(key);
}

CUSTOM public static bool SessionHasString(string key)
{
	SessionStrings::iterator i = s_SessionStrings.find(key);
	return i != s_SessionStrings.end();
}
