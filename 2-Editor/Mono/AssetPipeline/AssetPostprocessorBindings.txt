using System;
using UnityEngine;
using Object=UnityEngine.Object;
using System.Runtime.CompilerServices;
using System.Runtime.InteropServices;

namespace UnityEditor
{
// AssetPostprocessor lets you hook into the import pipeline and run scripts prior or after importing assets.


NONSEALED_CLASS AssetPostprocessor

	CSRAW private string m_PathName;

	// The path name of the asset being imported.
	CSRAW public string assetPath { get { return m_PathName; } set { m_PathName = value; } }
	
	// Logs an import warning to the console.
	CSRAW public void LogWarning (string warning, Object context = null) { Debug.LogWarning (warning, context); }
	
	// Logs an import error message to the console.
	CSRAW public void LogError (string warning, Object context = null) { Debug.LogError (warning, context); }

	// Returns the version of the asset postprocessor. 
	CSRAW public virtual uint GetVersion() {return 0;}

	// Reference to the asset importer
	CSRAW public AssetImporter assetImporter { get { return AssetImporter.GetAtPath(assetPath); }  }
	
	OBSOLETE error To set or get the preview, call EditorUtility.SetAssetPreview or AssetPreview.GetAssetPreview instead
	CSRAW public Texture2D preview { get { return null; } set { } }

	// Override the order in which importers are processed.
	CSRAW public virtual int GetPostprocessOrder () { return 0; }


	// Add this function in a subclass to get a notification just before the texture importer is run.
	CSNONE void OnPreprocessTexture ();

	// Add this function in a subclass to get a notification when a texture has completed importing just before
	CSNONE void OnPostprocessTexture (Texture2D texture);

	// Add this function in a subclass to get a notification just before a model (.fbx, .mb file etc.) is being imported.
	CSNONE void OnPreprocessModel ();

	// Add this function in a subclass to get a notification when a model has completed importing 
	CSNONE void OnPostprocessModel (GameObject root);

	// Gets called for each GameObject that had at least one userpropery attached to it in the imported file.
	CSNONE void OnPostprocessGameObjectWithUserProperties (GameObject root, string[] propNames, object[] values);

	// Feeds a source material 
	CSNONE Material OnAssignMaterialModel (Material material, Renderer renderer);


	// Add this function in a subclass to get a notification when an audio clip has completed importing.
	CSNONE void OnPostprocessAudio (AudioClip clip);

	// Add this function in a subclass to get a notification just before an audio clip is being imported.
	CSNONE void OnPreprocessAudio ();
 

	// This is called after importing of any number of assets is complete (when the Assets progress bar has reached the end).

	CSNONE static void OnPostprocessAllAssets (string[] importedAssets, string[] deletedAssets, string[] movedAssets, string[] movedFromPath);

END

CSRAW }
