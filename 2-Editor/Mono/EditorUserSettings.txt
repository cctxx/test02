CSRAW
using UnityEngine;

namespace UnityEditor
{

//*undocumented*
CLASS EditorUserSettings : Object

	CUSTOM static string GetConfigValue(string name)
	{
		if (!GetEditorUserSettings().HasConfigValue(name.AsUTF8()))
			return SCRIPTING_NULL;
		return scripting_string_new(GetEditorUserSettings().GetConfigValue(name.AsUTF8()));
	}

	CUSTOM static void SetConfigValue(string name, string value)
	{
		GetEditorUserSettings().SetConfigValue(name.AsUTF8(), value.AsUTF8());
	}
	
	CUSTOM_PROP static bool AutomaticAdd { return GetEditorUserSettings().GetVCAutomaticAdd(); } { GetEditorUserSettings().SetVCAutomaticAdd(value); }
	CUSTOM_PROP static bool WorkOffline { return GetEditorUserSettings().GetVCWorkOffline(); } { GetEditorUserSettings().SetVCWorkOffline(value); }
	CUSTOM_PROP internal static bool DebugCmd { return GetEditorUserSettings().GetVCDebugCmd(); } { GetEditorUserSettings().SetVCDebugCmd(value); }
	CUSTOM_PROP internal static bool DebugOut { return GetEditorUserSettings().GetVCDebugOut(); } { GetEditorUserSettings().SetVCDebugOut(value); }
	CUSTOM_PROP internal static bool DebugCom { return GetEditorUserSettings().GetVCDebugCom(); } { GetEditorUserSettings().SetVCDebugCom(value); }
END
