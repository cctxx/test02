using System;
using UnityEngine;

namespace UnityEditor
{

CLASS internal ColliderUtil

	CUSTOM static Vector3 GetCapsuleExtents(CapsuleCollider cc)
	{
		Vector2f vec =  cc->GetGlobalExtents();
		return Vector3f(vec.x, vec.y, 0.0f);
	}	

	CUSTOM static Matrix4x4 CalculateCapsuleTransform(CapsuleCollider cc)
	{
		return cc->CalculateTransform();
	}
	
END
